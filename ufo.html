Remove unnecessary data, seperate dates, and filter out anything not in the US.

ufo2 <- ufo %>%
  separate(datetime, into = c("day", "month","year"), sep="/") %>%
  separate(year, into = c("year", "time"), sep = " ") %>%
  select(-duration_hours_min, -comments, -date_posted) %>%
  filter(country=="us")
